  <script>
    const greetingBtn = document.getElementById("greetingBtn");
    const greetingMessage = document.getElementById("greetingMessage");
    const backgroundMusic = document.getElementById("backgroundMusic");
    let musicPlaying = false;

    greetingBtn.addEventListener("click", () => {
      greetingMessage.style.display = "block"; // Show greeting message
      greetingBtn.style.display = "none"; // Hide button after message shows
    });

    function toggleMusic() {
      if (musicPlaying) {
        backgroundMusic.pause();
        musicPlaying = false;
        document.getElementById("musicBtn").textContent = "Play Music";
      } else {
        backgroundMusic.play();
        musicPlaying = true;
        document.getElementById("musicBtn").textContent = "Pause Music";
      }
    }
